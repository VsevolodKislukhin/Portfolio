query = '''
SELECT t.author,
AVG(avg)
FROM
(SELECT b.book_id,
a.author_id,
a.author,
COUNT(ra.rating),
AVG(ra.rating)
FROM books AS b
LEFT JOIN authors AS a ON b.author_id = a.author_id
LEFT JOIN ratings AS ra ON b.book_id = ra.book_id
GROUP BY b.book_id,a.author_id
HAVING COUNT(ra.rating) >= 50) AS t
GROUP BY t.author
ORDER BY avg DESC
'''
output = pd.io.sql.read_sql(query, con = engine)
